def self.search_path=(keychains)
  search_list = keychains.map do |kc|
    "\"#{kc.path}\""
  end
  
  cmd = Xcode::Shell::Command.new "security"
  cmd << "list-keychain"
  cmd << "-s #{search_list.join(' ')}"
  cmd.execute
end
