def prepare_xcodebuild sdk=nil
  cmd = super sdk
  cmd << @scheme.parent.to_xcodebuild_option
  cmd << "-scheme \"#{@scheme.name}\""
  cmd << "-configuration \"#{@scheme.archive_config}\""
  cmd
end
