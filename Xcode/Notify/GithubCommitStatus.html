def initialize(builder, args={})
	origin_url = `git config --get remote.origin.url`.strip
	raise "Unable to parse github url #{origin_url}" unless origin_url=~/github\.com(\:?)\/([^\/]+)\/(.*)\.git$/
  # Default to Jenkin's build url env variable
  @target_url = args[:target_url]||ENV['BUILD_URL']
  @sha = args[:sha]||`git rev-parse HEAD`.strip
  @owner = args[:owner]||$2
  @repo = args[:repo]||$3
end

def perform state, description=nil
	RestClient.post "api.github.com/repos/#{@owner}/#{@repo}/statuses/#{@sha}", {
		:state => state,
		:target_url => @target_url,
		:description => description
	}
end
